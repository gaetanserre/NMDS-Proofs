/-
  We provide a typeclass definition for a generic RKHS.
-/

class RKHS {E F : Type*} [RCLike F] (H : Set (E → F)) [NormedAddCommGroup H] [InnerProductSpace F H] where
  k : E → E → F
  memb : ∀ (x : E), k x ∈ H
  repro : ∀ (hf : f ∈ H), ∀ (x : E), f x = inner (⟨f, hf⟩ : H) ⟨k x, memb x⟩
